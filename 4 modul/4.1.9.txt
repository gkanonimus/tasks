/* Распечатать первые n простых чисел */

#include <stdio.h>

int main()
{
    int n, count, number, divider, temp_count;
    n = 10;
    count = 0;
    
    for(number = 2; n > count; number++)
    {
        for(divider = 1; divider != number; divider++)
            if (number % divider == 0)
    	        temp_count++;
    	    if (temp_count == 1)
    	    {
    	        printf("\n%d\n", number);
    	        count++;
    	    }
    	temp_count = 0;
    }
    
}
