/* «Упаковать» четыре символа в 32-битное беззнаковое целое, затем «распаковать» 32-битное беззнаковое целое число в четыре символа */

#include <stdio.h>

int main()
{
    unsigned int x;
    char a, b, c, d;
    a = 'g';   // g = 0110 0111
    b = '7';   // 7 = 0011 0111
    c = 'K';   // K = 0100 1011
    d = '+';   // + = 0010 1011
    x = (((a << 8) + b << 8) + c << 8) + d;
    printf("\n%u\n", x);
    
    a = x >> 24;
    b = x << 8 >> 24;
    c = x >> 8 % 16;
    d = x % 128;
    printf("\n%c\t%c\t%c\t%c\n", a, b, c, d);
}
