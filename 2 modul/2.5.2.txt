/* Целой переменной k присвоить значение, равное сумме цифр в записи целого
положительного трехзначного числа x */
#include <stdio.h>
#include <math.h>

int main()
{
    int k, x;
    x = 374;
    k = 0;
    for (int i = 0; i < 3; i++)
    	{
    	k += x / (int)pow(10, i) % 10;
    	}
    printf("\n%d\n", k);
}
